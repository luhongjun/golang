# 策略模式

策略行为设计模式允许在运行时选择算法的行为。
【注意注意】是运行时选择算法！！！ 策略模式允许您更改对象的内部

它定义算法，封装算法，并互换使用算法。

```go
package main

import "fmt"

type Operator interface {
	Apply(int, int) int
}

type Operation struct {
	operator Operator
}

func (operation *Operation) SetOperator(operator Operator) *Operation {
	operation.operator = operator
	return operation
}

func (operation *Operation) Operate(leftValue, rightValue int) int {
	return operation.operator.Apply(leftValue, rightValue)
}

// Addition 加法算法
type Addition struct{}
func (Addition) Apply(lval, rval int) int {
	return lval + rval
}

// Multiplication 乘法算法
type Multiplication struct{}
func (Multiplication) Apply(lval, rval int) int {
	return lval * rval
}

func main() {
	operation := Operation{}
	
	fmt.Println(operation.SetOperator(Addition{}).Operate(3, 5)) // 8

	fmt.Println(operation.SetOperator(Multiplication{}).Operate(3, 5)) // 15
}
```
